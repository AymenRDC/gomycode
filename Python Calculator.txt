from google.colab import output
import time

quitter = False
while(quitter == False):

  print("""
choose an operation :
  1-addition
  2-substraction
  3-multiplication
  4-division
  5-exit""")
# user will choose the operation
  operation = int(input("your choice : "))
# user will choose the numbers
  n1 = float(input("give me a number : "))
  n2 = float(input("give me a number : "))

  if operation == 1:
  #addition
    res = n1 + n2
    print("the result is : ", res)
  elif operation == 2:
  #substruction
    res = n1 - n2
    print("the result is : ", res)
  elif operation == 3:
  #multiplication
    res = n1 * n2
    print("the result is : ", res)
  elif operation == 4:
  #division
    res = n1 / n2
    print("the result is : ", res)
  elif operation == 5:
    # exit the calculator
    quitter = True
  else :
    print("wrong choice, try again please")
  time.sleep(4)
  output.clear()
